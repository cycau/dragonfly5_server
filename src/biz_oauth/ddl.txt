CREATE TABLE d5_member (
    id VARCHAR(20) PRIMARY KEY,
    user_role VARCHAR(8),
    email VARCHAR(100),
    email_verified BOOLEAN,
    status VARCHAR(4),
    password VARCHAR(200),
    nickname VARCHAR(100),
    avatar VARCHAR(200),
    allowed_wrong_time INT, // random 3 ~ 7 - prev_locked_count, when wrong to set
    lock_until TIMESTAMPTZ, // one huour, when wrong to set
    prev_locked_count INT,  // 2^prev_locked_count 5min
    last_logged_in_at TIMESTAMPTZ,
    created_at TIMESTAMPTZ,
    updated_at TIMESTAMPTZ
);
